    if (rightInfo != null) {

    // è¿™äº›æƒ…å†µæœ‰å“ªäº›å‘¢ï¼Ÿ
    // 1. å·¦æ ‘æˆ–å³æ ‘æœ‰å€¼çš„æƒ…å†µï¼šçœ‹å·¦æ ‘å’Œå³æ ‘æ˜¯å¦æ»¡è¶³æœç´¢æ ‘çš„æ¡ä»¶ã€çœ‹å€¼æ˜¯å¦ç¬¦åˆæ¡ä»¶ã€‚
    // 2. å·¦æ ‘å’Œå³æ ‘éƒ½æ˜¯nullï¼Œç›´æ¥è¿”å›trueã€‚
    boolean isBST = true;
    if (leftInfo != null && !leftInfo.isBST) {

    if (rightInfo != null && !leftInfo.isBST) {
        isBST = false;
    }
    // åˆ¤è¯»å·¦æ ‘çš„æœ€å¤§å€¼æ˜¯å¦å°äºrootçš„å€¼ã€‚
    boolean leftMaxLessX = leftInfo == null ? true : (leftInfo.max < root.val);

    boolean rightMinMoreX = leftInfo == null ? true : (rightInfo.min > root.val);

    if (!leftMaxLessX || !rightMinMoreX) {
        isBST = false;
    }

    return new Info(isBST, max, min);
}

public static void process(TreeNode x, List<Integer> path, int preSum, int sum, List<List<Integer>> ans) {
    // basecaseå¶èŠ‚ç‚¹
    if (x.left == null && x.right == null) {
        if (preSum + x.val == sum) {
            path.add(x.val);
            ans.add(copy(path));
            path.remove(path.size() - 1);
        }
        return;
    }

    preSum += x.val;
    if (x.left != null) {
        process(x.left, path, preSum, sum, ans);
    }
    if (x.right != null) {
        process(x.right, path, preSum, sum, ans);
    }
    path.remove(path.size() - 1);
}

public static List<Integer> copy(List<Integer> path) {
    List<Integer> ans = new ArrayList<>();
    for (Integer num : path) {
        ans.add(num);
